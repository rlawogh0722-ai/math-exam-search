function doGet(e) { return ContentService .createTextOutput(JSON.stringify(handleRequest(e))) .setMimeType(ContentService.MimeType.JSON); } function handleRequest(e) { const query = e.parameter.q || ''; if (!query) return []; const keywords = query.toLowerCase().split(' ').filter(k => k.trim()); const folderId = getFolderId('기출_PDF'); if (!folderId) return { error: '기출_PDF 폴더를 찾을 수 없습니다.' }; const folder = DriveApp.getFolderById(folderId); const files = folder.getFilesByType('application/pdf'); const results = []; while (files.hasNext()) { const file = files.next(); const fname = file.getName().replace('.pdf', '').toLowerCase(); const normalized = fname.normalize('NFC'); if (keywords.every(kw => normalized.includes(kw.normalize('NFC')))) { const parts = file.getName().replace('.pdf', '').split('_'); results.push({ name: file.getName(), link: file.getUrl(), major: parts[0] || '', mid: parts[1] || '', minor: parts[2] || '', year: parts[3] || '' }); } } return results; } function getFolderId(folderName) { const folders = DriveApp.getFoldersByName(folderName); return folders.hasNext() ? folders.next().getId() : null; }
